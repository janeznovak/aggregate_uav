73 // A string-managing class for DLL boundary safe text.
74 // It ensures that strings are de-allocated with the same
75 // memory manager they are made with. This allows you to use a debug
76 // build of your client with a release SDK.
77 class String
78 {
79 public:
80  // A string which we are not responsible for deallocating
81  inline
82  String( const char * i_pString = 0 )
83  : m_pString( 0 )
84  , m_pConstString( i_pString )
85  , m_pStringFactory( 0 )
86  {
87  }
88 
89  // A string which we are not responsible for deallocating
90  String( const std::string & i_rString )
91  : m_pString( 0 )
92  , m_pConstString( i_rString.c_str() )
93  , m_pStringFactory( 0 )
94  {
95  }
96 
97  // Copy constructor
98  inline
99  String( const String & i_rString )
100  {
101  m_pConstString = i_rString.m_pConstString;
102  m_pStringFactory = i_rString.m_pStringFactory;
103  if( m_pStringFactory )
104  {
105  m_pString = m_pStringFactory->AllocAndCopyString( i_rString.m_pString );
106  }
107  else
108  {
109  m_pString = 0;
110  }
111  }
112 
113  inline
114  String & operator=( const String & i_rString )
115  {
116  m_pConstString = i_rString.m_pConstString;
117  m_pStringFactory = i_rString.m_pStringFactory;
118  if ( m_pStringFactory )
119  {
120  m_pString = m_pStringFactory->AllocAndCopyString( i_rString.m_pString );
121  }
122  else
123  {
124  m_pString = 0;
125  }
126  return *this;
127  }
128 
129  inline
130  ~String()
131  {
132  if( m_pStringFactory )
133  {
134  m_pStringFactory->FreeString( m_pString );
135  }
136  }
137 
138  // A string which we are responsible for deallocating
139  inline
140  void Set( const char * i_pString, IStringFactory & i_rStringFactory )
141  {
142  m_pString = i_rStringFactory.AllocAndCopyString( i_pString );
143  m_pStringFactory = &i_rStringFactory;
144  m_pConstString = 0;
145  }
146 
147  inline
148  operator std::string() const
149  {
150  if( m_pStringFactory )
151  {
152  return std::string( m_pString );
153  }
154  else
155  {
156  return std::string( m_pConstString );
157  }
158  }
159 
160 private:
161  char * m_pString;
162  const char * m_pConstString;
163  IStringFactory * m_pStringFactory;
164 };
