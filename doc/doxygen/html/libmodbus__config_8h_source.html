199 /*
200  * Copy src to string dst of size siz. At most siz-1 characters
201  * will be copied. Always NUL terminates (unless siz == 0).
202  * Returns strlen(src); if retval >= siz, truncation occurred.
203  */
204 static size_t
205 strlcpy(char *dst, const char *src, size_t siz)
206 {
207  char *d = dst;
208  const char *s = src;
209  size_t n = siz;
210 
211  /* Copy as many bytes as will fit */
212  if (n != 0) {
213  while (--n != 0) {
214  if ((*d++ = *s++) == '\0')
215  break;
216  }
217  }
218 
219  /* Not enough room in dst, add NUL and traverse rest of src */
220  if (n == 0) {
221  if (siz != 0)
222  *d = '\0'; /* NUL-terminate dst */
223  while (*s++)
224  ;
225  }
226 
227  return(s - src - 1); /* count does not include NUL */
228 }
