683  old(CALL, robot_phase::IDLE, [&](robot_phase ph) {
684  std::string rname = ROBOT_PREFIX + std::to_string(node.uid); // name is obtained from node ID
685  node.storage(node_external_name{}) = rname;
686 
687  std::lock_guard lgr(RobotStatesMutex);
688  if (RobotStatesMap.find(rname) != RobotStatesMap.end()) {
689  for (RobotStatus rs : RobotStatesMap[rname]) {
690  // add offset to match with background
691  real_t pos_x_offset = node.storage(node_offset_pos_x{});
692  real_t pos_y_offset = node.storage(node_offset_pos_y{});
693  node.position() = make_vec(rs.pos_x + pos_x_offset, rs.pos_y + pos_y_offset, rs.pos_z);
694 
695  // truncate percentage to simply the output: 98 -> 90; 92 -> 90; 85 -> 80
696  float battery_percent_charge_trunc = std::trunc(rs.battery_percent_charge / 10) * 10;
697  if (rs.battery_percent_charge >= 0) {
698  // resize the node according with the battery percent charge
699  node.storage(node_size{}) = (battery_percent_charge_trunc / 100.0) * NODE_SIZE;
700  node.storage(node_shadow_size{}) = (battery_percent_charge_trunc / 100.0) * NODE_SHADOW_SIZE;
701  }
702  else {
703  node.storage(node_size{}) = NODE_SIZE / 2;
704  node.storage(node_shadow_size{}) = NODE_SHADOW_SIZE / 2;
705  }
706 
707  // change colour according with feedback from robots
708  fcpp::color new_color = fcpp::color(fcpp::coordination::idle_color);
709  // if new state is RUNNING: change to "running color"
710  if (feedback::GoalStatus::RUNNING == rs.goal_status) {
711  manage_running_goal_status(CALL);
712  }
713  // if new state is REACHED:
714  else if (feedback::GoalStatus::REACHED == rs.goal_status) {
715  manage_reached_goal_status(CALL);
716  }
717  // if new state is UNKNOWN: new color is idle
718  else if (feedback::GoalStatus::UNKNOWN == rs.goal_status) {
719  manage_unknown_goal_status(CALL);
720  }
721  // if new state is FAILED: new color is failed
722  else if (feedback::GoalStatus::FAILED == rs.goal_status) {
723  manage_failed_goal_status(CALL);
724  }
725  // if new state is ABORTED: new color is aborted
726  else if (feedback::GoalStatus::ABORTED == rs.goal_status) {
727  manage_aborted_goal_status(CALL);
728  }
729  // if new state is NO GOAL: new color is idle
730  else if (feedback::GoalStatus::NO_GOAL == rs.goal_status) {
731  manage_no_goal_status(CALL);
732  }
733  // otherwise: maintain previous
734  else {

736  }
737

740  // update external goal in storage
741  node.storage(node_external_goal{}) = rs.goal_code;
742  node.storage(node_battery_charge{}) = battery_percent_charge_trunc;
743

746  manage_battery_discharged_node(CALL);
747  }
748  }
749  // delete element from map

751  }

765  field<int> test_nbr = fcpp::coordination::nbr(CALL, n_round);

772  node.storage(node_process_goal{}) == common::get<goal_code>(g) &&
773  node.storage(node_external_goal{}) == common::get<goal_code>(g)) {
774  manage_action_abort(CALL, g, &s);
775  }
776

802  FUN void manage_termination(ARGS, node_type nt, spawn_result_type& r) {
803  // if process was terminating and now it's terminated, we have to change state machine to IDLE
804  if (node.storage(node_process_status{}) == ProcessingStatus::TERMINATING) {
805  // if process has been terminated, it isn't in the result map of spawn
806  bool process_found = false;
807  for (auto const& x : r) {
808  auto g = x.first;
809  if (common::get<goal_code>(g) == node.storage(node_process_goal{})) {
810  process_found = true;
811  }
812  }
813  if (!process_found) {
814  std::cout << "Process with code " << node.storage(node_process_goal{}) << " not found, so move robot to IDLE" << endl;
815  node.storage(node_process_status{}) = ProcessingStatus::IDLE;
816  }
817  }
818 
819  // search on results if the computing processes has been terminated:
820  // - if it's terminated (or in other words, if the goal it's not in the "r" variable), we delete it from the map in the storage
821  std::vector<std::string> goals_to_remove = {};
822  for (auto const& x : node.storage(node_process_computing_goals{}))
823  {
824  auto g = x.second;
825  if (r.find(g) == r.end()) {
826  std::cout << "Remove process with code " << common::get<goal_code>(g) << endl;
827  goals_to_remove.push_back(common::get<goal_code>(g));
828  }
829  }
830  for (auto const& gc : goals_to_remove) {
831  remove_goal_from_computing_map(CALL, gc);
832  }
833  }
