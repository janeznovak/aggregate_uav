90 // Captures numpy/eigen conformability status (returned by EigenProps::conformable()):
91 template <bool EigenRowMajor>
92 struct EigenConformable {
93  bool conformable = false;
94  EigenIndex rows = 0, cols = 0;
95  EigenDStride stride{0, 0}; // Only valid if negativestrides is false!
96  bool negativestrides = false; // If true, do not use stride!
97 
98  // NOLINTNEXTLINE(google-explicit-constructor)
99  EigenConformable(bool fits = false) : conformable{fits} {}
100  // Matrix type:
101  EigenConformable(EigenIndex r, EigenIndex c, EigenIndex rstride, EigenIndex cstride)
102  : conformable{true}, rows{r}, cols{c},
103  // TODO: when Eigen bug #747 is fixed, remove the tests for non-negativity.
104  // http://eigen.tuxfamily.org/bz/show_bug.cgi?id=747
105  stride{EigenRowMajor ? (rstride > 0 ? rstride : 0)
106  : (cstride > 0 ? cstride : 0) /* outer stride */,
107  EigenRowMajor ? (cstride > 0 ? cstride : 0)
108  : (rstride > 0 ? rstride : 0) /* inner stride */},
109  negativestrides{rstride < 0 || cstride < 0} {}
110  // Vector type:
111  EigenConformable(EigenIndex r, EigenIndex c, EigenIndex stride)
112  : EigenConformable(r, c, r == 1 ? c * stride : stride, c == 1 ? r : r * stride) {}
113 
114  template <typename props>
115  bool stride_compatible() const {
116  // To have compatible strides, we need (on both dimensions) one of fully dynamic strides,
117  // matching strides, or a dimension size of 1 (in which case the stride value is
118  // irrelevant). Alternatively, if any dimension size is 0, the strides are not relevant
119  // (and numpy â‰¥ 1.23 sets the strides to 0 in that case, so we need to check explicitly).
120  if (negativestrides) {
121  return false;
122  }
123  if (rows == 0 || cols == 0) {
124  return true;
125  }
126  return (props::inner_stride == Eigen::Dynamic || props::inner_stride == stride.inner()
127  || (EigenRowMajor ? cols : rows) == 1)
128  && (props::outer_stride == Eigen::Dynamic || props::outer_stride == stride.outer()
129  || (EigenRowMajor ? rows : cols) == 1);
130  }
131  // NOLINTNEXTLINE(google-explicit-constructor)
132  operator bool() const { return conformable; }
133 };

148 // Helper struct for extracting information from an Eigen type
149 template <typename Type_>
150 struct EigenProps {
151  using Type = Type_;
152  using Scalar = typename Type::Scalar;
153  using StrideType = typename eigen_extract_stride<Type>::type;
154  static constexpr EigenIndex rows = Type::RowsAtCompileTime, cols = Type::ColsAtCompileTime,
155  size = Type::SizeAtCompileTime;
156  static constexpr bool row_major = Type::IsRowMajor,
157  vector
158  = Type::IsVectorAtCompileTime, // At least one dimension has fixed size 1
159  fixed_rows = rows != Eigen::Dynamic, fixed_cols = cols != Eigen::Dynamic,
160  fixed = size != Eigen::Dynamic, // Fully-fixed size
161  dynamic = !fixed_rows && !fixed_cols; // Fully-dynamic size
162 
163  template <EigenIndex i, EigenIndex ifzero>
164  using if_zero = std::integral_constant<EigenIndex, i == 0 ? ifzero : i>;
165  static constexpr EigenIndex inner_stride
166  = if_zero<StrideType::InnerStrideAtCompileTime, 1>::value,
167  outer_stride = if_zero < StrideType::OuterStrideAtCompileTime,
168  vector ? size
169  : row_major ? cols
170  : rows > ::value;
171  static constexpr bool dynamic_stride
172  = inner_stride == Eigen::Dynamic && outer_stride == Eigen::Dynamic;
173  static constexpr bool requires_row_major
174  = !dynamic_stride && !vector && (row_major ? inner_stride : outer_stride) == 1;
175  static constexpr bool requires_col_major
176  = !dynamic_stride && !vector && (row_major ? outer_stride : inner_stride) == 1;

178  // Takes an input array and determines whether we can make it fit into the Eigen type. If
179  // the array is a vector, we attempt to fit it into either an Eigen 1xN or Nx1 vector
180  // (preferring the latter if it will fit in either, i.e. for a fully dynamic matrix type).
181  static EigenConformable<row_major> conformable(const array &a) {
182  const auto dims = a.ndim();
183  if (dims < 1 || dims > 2) {
184  return false;
185  }
186 
187  if (dims == 2) { // Matrix type: require exact match (or dynamic)
188 
189  EigenIndex np_rows = a.shape(0), np_cols = a.shape(1),
190  np_rstride = a.strides(0) / static_cast<ssize_t>(sizeof(Scalar)),
191  np_cstride = a.strides(1) / static_cast<ssize_t>(sizeof(Scalar));
192  if ((PYBIND11_SILENCE_MSVC_C4127(fixed_rows) && np_rows != rows)
193  || (PYBIND11_SILENCE_MSVC_C4127(fixed_cols) && np_cols != cols)) {
194  return false;
195  }
196 
197  return {np_rows, np_cols, np_rstride, np_cstride};
198  }
199 
200  // Otherwise we're storing an n-vector. Only one of the strides will be used, but
201  // whichever is used, we want the (single) numpy stride value.
202  const EigenIndex n = a.shape(0),
203  stride = a.strides(0) / static_cast<ssize_t>(sizeof(Scalar));
204 
205  if (vector) { // Eigen type is a compile-time vector
206  if (PYBIND11_SILENCE_MSVC_C4127(fixed) && size != n) {
207  return false; // Vector size mismatch
208  }
209  return {rows == 1 ? 1 : n, cols == 1 ? 1 : n, stride};
210  }
211  if (fixed) {
212  // The type has a fixed size, but is not a vector: abort
213  return false;
214  }
215  if (fixed_cols) {
216  // Since this isn't a vector, cols must be != 1. We allow this only if it exactly
217  // equals the number of elements (rows is Dynamic, and so 1 row is allowed).
218  if (cols != n) {
219  return false;
220  }
221  return {1, n, stride};
222  } // Otherwise it's either fully dynamic, or column dynamic; both become a column vector
223  if (PYBIND11_SILENCE_MSVC_C4127(fixed_rows) && rows != n) {
224  return false;
225  }
226  return {n, 1, stride};
227  }

252 // Casts an Eigen type to numpy array. If given a base, the numpy array references the src data,
253 // otherwise it'll make a copy. writeable lets you turn off the writeable flag for the array.
254 template <typename props>
255 handle
256 eigen_array_cast(typename props::Type const &src, handle base = handle(), bool writeable = true) {
257  constexpr ssize_t elem_size = sizeof(typename props::Scalar);
258  array a;
259  if (props::vector) {
260  a = array({src.size()}, {elem_size * src.innerStride()}, src.data(), base);
261  } else {
262  a = array({src.rows(), src.cols()},
263  {elem_size * src.rowStride(), elem_size * src.colStride()},
264  src.data(),
265  base);
266  }
267 
268  if (!writeable) {
269  array_proxy(a.ptr())->flags &= ~detail::npy_api::NPY_ARRAY_WRITEABLE_;
270  }
271 
272  return a.release();
273 }

275 // Takes an lvalue ref to some Eigen type and a (python) base object, creating a numpy array that
276 // reference the Eigen object's data with `base` as the python-registered base class (if omitted,
277 // the base will be set to None, and lifetime management is up to the caller). The numpy array is
278 // non-writeable if the given type is const.
279 template <typename props, typename Type>
280 handle eigen_ref_array(Type &src, handle parent = none()) {
281  // none here is to get past array's should-we-copy detection, which currently always
282  // copies when there is no base. Setting the base to None should be harmless.
283  return eigen_array_cast<props>(src, parent, !std::is_const<Type>::value);
284 }

286 // Takes a pointer to some dense, plain Eigen type, builds a capsule around it, then returns a
287 // numpy array that references the encapsulated data with a python-side reference to the capsule to
288 // tie its destruction to that of any dependent python objects. Const-ness is determined by
289 // whether or not the Type of the pointer given is const.
290 template <typename props, typename Type, typename = enable_if_t<is_eigen_dense_plain<Type>::value>>
291 handle eigen_encapsulate(Type *src) {
292  capsule base(src, [](void *o) { delete static_cast<Type *>(o); });
293  return eigen_ref_array<props>(*src, base);
294 }

296 // Type caster for regular, dense matrix types (e.g. MatrixXd), but not maps/refs/etc. of dense
297 // types.
298 template <typename Type>
299 struct type_caster<Type, enable_if_t<is_eigen_dense_plain<Type>::value>> {
300  using Scalar = typename Type::Scalar;
301  using props = EigenProps<Type>;
302 
303  bool load(handle src, bool convert) {
304  // If we're in no-convert mode, only load if given an array of the correct type
305  if (!convert && !isinstance<array_t<Scalar>>(src)) {
306  return false;
307  }
308 
309  // Coerce into an array, but don't do type conversion yet; the copy below handles it.
310  auto buf = array::ensure(src);
311 
312  if (!buf) {
313  return false;
314  }
315 
316  auto dims = buf.ndim();
317  if (dims < 1 || dims > 2) {
318  return false;
319  }
320 
321  auto fits = props::conformable(buf);
322  if (!fits) {
323  return false;
324  }
325 
326  // Allocate the new type, then build a numpy reference into it
327  value = Type(fits.rows, fits.cols);
328  auto ref = reinterpret_steal<array>(eigen_ref_array<props>(value));
329  if (dims == 1) {
330  ref = ref.squeeze();
331  } else if (ref.ndim() == 1) {
332  buf = buf.squeeze();
333  }
334 
335  int result = detail::npy_api::get().PyArray_CopyInto_(ref.ptr(), buf.ptr());
336 
337  if (result < 0) { // Copy failed!
338  PyErr_Clear();
339  return false;
340  }
341 
342  return true;
343  }
344 
345 private:
346  // Cast implementation
347  template <typename CType>
348  static handle cast_impl(CType *src, return_value_policy policy, handle parent) {
349  switch (policy) {
350  case return_value_policy::take_ownership:
351  case return_value_policy::automatic:
352  return eigen_encapsulate<props>(src);
353  case return_value_policy::move:
354  return eigen_encapsulate<props>(new CType(std::move(*src)));
355  case return_value_policy::copy:
356  return eigen_array_cast<props>(*src);
357  case return_value_policy::reference:
358  case return_value_policy::automatic_reference:
359  return eigen_ref_array<props>(*src);
360  case return_value_policy::reference_internal:
361  return eigen_ref_array<props>(*src, parent);
362  default:
363  throw cast_error("unhandled return_value_policy: should not happen!");
364  };
365  }
366 
367 public:
368  // Normal returned non-reference, non-const value:
369  static handle cast(Type &&src, return_value_policy /* policy */, handle parent) {
370  return cast_impl(&src, return_value_policy::move, parent);
371  }
372  // If you return a non-reference const, we mark the numpy array readonly:
373  static handle cast(const Type &&src, return_value_policy /* policy */, handle parent) {
374  return cast_impl(&src, return_value_policy::move, parent);
375  }
376  // lvalue reference return; default (automatic) becomes copy
377  static handle cast(Type &src, return_value_policy policy, handle parent) {
378  if (policy == return_value_policy::automatic
379  || policy == return_value_policy::automatic_reference) {
380  policy = return_value_policy::copy;
381  }
382  return cast_impl(&src, policy, parent);
383  }
384  // const lvalue reference return; default (automatic) becomes copy
385  static handle cast(const Type &src, return_value_policy policy, handle parent) {
386  if (policy == return_value_policy::automatic
387  || policy == return_value_policy::automatic_reference) {
388  policy = return_value_policy::copy;
389  }
390  return cast(&src, policy, parent);
391  }
392  // non-const pointer return
393  static handle cast(Type *src, return_value_policy policy, handle parent) {
394  return cast_impl(src, policy, parent);
395  }
396  // const pointer return
397  static handle cast(const Type *src, return_value_policy policy, handle parent) {
398  return cast_impl(src, policy, parent);
399  }

405  // NOLINTNEXTLINE(google-explicit-constructor)

407  // NOLINTNEXTLINE(google-explicit-constructor)

417 template <typename MapType>
418 struct eigen_map_caster {
419 private:
420  using props = EigenProps<MapType>;
421 
422 public:
423  // Directly referencing a ref/map's data is a bit dangerous (whatever the map/ref points to has
424  // to stay around), but we'll allow it under the assumption that you know what you're doing
425  // (and have an appropriate keep_alive in place). We return a numpy array pointing directly at
426  // the ref's data (The numpy array ends up read-only if the ref was to a const matrix type.)

430  switch (policy) {
431  case return_value_policy::copy:

433  case return_value_policy::reference_internal:
434  return eigen_array_cast<props>(src, parent, is_eigen_mutable_map<MapType>::value);
435  case return_value_policy::reference:
436  case return_value_policy::automatic:
437  case return_value_policy::automatic_reference:
438  return eigen_array_cast<props>(src, none(), is_eigen_mutable_map<MapType>::value);
439  default:
440  // move, take_ownership don't make any sense for a ref/map:
441  pybind11_fail("Invalid return_value_policy for Eigen Map/Ref/Block type");

460 // Loader for Ref<...> arguments. See the documentation for info on how to make this work without
461 // copying (it requires some extra effort in many cases).
462 template <typename PlainObjectType, typename StrideType>
463 struct type_caster<
464  Eigen::Ref<PlainObjectType, 0, StrideType>,
465  enable_if_t<is_eigen_dense_map<Eigen::Ref<PlainObjectType, 0, StrideType>>::value>>
466  : public eigen_map_caster<Eigen::Ref<PlainObjectType, 0, StrideType>> {
467 private:
468  using Type = Eigen::Ref<PlainObjectType, 0, StrideType>;
469  using props = EigenProps<Type>;
470  using Scalar = typename props::Scalar;
471  using MapType = Eigen::Map<PlainObjectType, 0, StrideType>;
472  using Array
473  = array_t<Scalar,
474  array::forcecast
475  | ((props::row_major ? props::inner_stride : props::outer_stride) == 1
476  ? array::c_style
477  : (props::row_major ? props::outer_stride : props::inner_stride) == 1
478  ? array::f_style
479  : 0)>;
480  static constexpr bool need_writeable = is_eigen_mutable_map<Type>::value;
481  // Delay construction (these have no default constructor)
482  std::unique_ptr<MapType> map;
483  std::unique_ptr<Type> ref;
484  // Our array. When possible, this is just a numpy array pointing to the source data, but
485  // sometimes we can't avoid copying (e.g. input is not a numpy array at all, has an
486  // incompatible layout, or is an array of a type that needs to be converted). Using a numpy
487  // temporary (rather than an Eigen temporary) saves an extra copy when we need both type
488  // conversion and storage order conversion. (Note that we refuse to use this temporary copy
489  // when loading an argument for a Ref<M> with M non-const, i.e. a read-write reference).
490  Array copy_or_ref;
491 
492 public:
493  bool load(handle src, bool convert) {
494  // First check whether what we have is already an array of the right type. If not, we
495  // can't avoid a copy (because the copy is also going to do type conversion).
496  bool need_copy = !isinstance<Array>(src);
497 
498  EigenConformable<props::row_major> fits;
499  if (!need_copy) {
500  // We don't need a converting copy, but we also need to check whether the strides are
501  // compatible with the Ref's stride requirements
502  auto aref = reinterpret_borrow<Array>(src);
503 
504  if (aref && (!need_writeable || aref.writeable())) {
505  fits = props::conformable(aref);
506  if (!fits) {
507  return false; // Incompatible dimensions
508  }
509  if (!fits.template stride_compatible<props>()) {
510  need_copy = true;
511  } else {
512  copy_or_ref = std::move(aref);
513  }
514  } else {
515  need_copy = true;
516  }
517  }
518 
519  if (need_copy) {
520  // We need to copy: If we need a mutable reference, or we're not supposed to convert
521  // (either because we're in the no-convert overload pass, or because we're explicitly
522  // instructed not to copy (via `py::arg().noconvert()`) we have to fail loading.
523  if (!convert || need_writeable) {
524  return false;
525  }
526 
527  Array copy = Array::ensure(src);
528  if (!copy) {
529  return false;
530  }
531  fits = props::conformable(copy);
532  if (!fits || !fits.template stride_compatible<props>()) {
533  return false;
534  }
535  copy_or_ref = std::move(copy);
536  loader_life_support::add_patient(copy_or_ref);
537  }
538 
539  ref.reset();
540  map.reset(new MapType(data(copy_or_ref),
541  fits.rows,
542  fits.cols,
543  make_stride(fits.stride.outer(), fits.stride.inner())));
544  ref.reset(new Type(*map));
545 
546  return true;
