111 // Python loads modules by default with dlopen with the RTLD_LOCAL flag; under libc++ and possibly
112 // other STLs, this means `typeid(A)` from one module won't equal `typeid(A)` from another module
113 // even when `A` is the same, non-hidden-visibility type (e.g. from a common include). Under
114 // libstdc++, this doesn't happen: equality and the type_index hash are based on the type name,
115 // which works. If not under a known-good stl, provide our own name-based hash and equality
116 // functions that use the type name.

349 inline void translate_exception(std::exception_ptr p) {
350  if (!p) {
351  return;
352  }
353  try {
354  std::rethrow_exception(p);
355  } catch (error_already_set &e) {
356  handle_nested_exception(e, p);
357  e.restore();
358  return;
359  } catch (const builtin_exception &e) {
360  // Could not use template since it's an abstract class.
361  if (const auto *nep = dynamic_cast<const std::nested_exception *>(std::addressof(e))) {
362  handle_nested_exception(*nep, p);
363  }
364  e.set_error();
365  return;
366  } catch (const std::bad_alloc &e) {
367  handle_nested_exception(e, p);
368  raise_err(PyExc_MemoryError, e.what());
369  return;
370  } catch (const std::domain_error &e) {
371  handle_nested_exception(e, p);
372  raise_err(PyExc_ValueError, e.what());
373  return;
374  } catch (const std::invalid_argument &e) {
375  handle_nested_exception(e, p);
376  raise_err(PyExc_ValueError, e.what());
377  return;
378  } catch (const std::length_error &e) {
379  handle_nested_exception(e, p);
380  raise_err(PyExc_ValueError, e.what());
381  return;
382  } catch (const std::out_of_range &e) {
383  handle_nested_exception(e, p);
384  raise_err(PyExc_IndexError, e.what());
385  return;
386  } catch (const std::range_error &e) {
387  handle_nested_exception(e, p);
388  raise_err(PyExc_ValueError, e.what());
389  return;
390  } catch (const std::overflow_error &e) {
391  handle_nested_exception(e, p);
392  raise_err(PyExc_OverflowError, e.what());
393  return;
394  } catch (const std::exception &e) {
395  handle_nested_exception(e, p);
396  raise_err(PyExc_RuntimeError, e.what());
397  return;
398  } catch (const std::nested_exception &e) {
399  handle_nested_exception(e, p);
400  raise_err(PyExc_RuntimeError, "Caught an unknown nested exception!");
401  return;
402  } catch (...) {
403  raise_err(PyExc_RuntimeError, "Caught an unknown exception!");
404  return;
405  }
406 }

436  // Cannot use py::gil_scoped_acquire here since that constructor calls get_internals.

477  }
478  PYBIND11_TLS_REPLACE_VALUE(internals_ptr->tstate, tstate);
479 
480 # if PYBIND11_INTERNALS_VERSION > 4
481  if (!PYBIND11_TLS_KEY_CREATE(internals_ptr->loader_life_support_tls_key)) {
482  pybind11_fail("get_internals: could not successfully initialize the "
483  "loader_life_support TSS key!");
484  }
485 # endif
486  internals_ptr->istate = tstate->interp;
487 #endif
488  builtins[id] = capsule(internals_pp);
489  internals_ptr->registered_exception_translators.push_front(&translate_exception);
490  internals_ptr->static_property_type = make_static_property_type();
491  internals_ptr->default_metaclass = make_default_metaclass();
492  internals_ptr->instance_base = make_object_base_type(internals_ptr->default_metaclass);
493  }
494  return **internals_pp;
495 }

497 // the internals struct (above) is shared between all the modules. local_internals are only
498 // for a single module. Any changes made to internals may require an update to
499 // PYBIND11_INTERNALS_VERSION, breaking backwards compatibility. local_internals is, by design,
500 // restricted to a single module. Whether a module has local internals or not should not
501 // impact any other modules, because the only things accessing the local internals is the
502 // module that contains them.
503 struct local_internals {
504  type_map<type_info *> registered_types_cpp;
505  std::forward_list<ExceptionTranslator> registered_exception_translators;
506 #if defined(WITH_THREAD) && PYBIND11_INTERNALS_VERSION == 4
507 
508  // For ABI compatibility, we can't store the loader_life_support TLS key in
509  // the `internals` struct directly. Instead, we store it in `shared_data` and
510  // cache a copy in `local_internals`. If we allocated a separate TLS key for
511  // each instance of `local_internals`, we could end up allocating hundreds of
512  // TLS keys if hundreds of different pybind11 modules are loaded (which is a
513  // plausible number).
514  PYBIND11_TLS_KEY_INIT(loader_life_support_tls_key)
515 
516  // Holds the shared TLS key for the loader_life_support stack.
517  struct shared_loader_life_support_data {
518  PYBIND11_TLS_KEY_INIT(loader_life_support_tls_key)
519  shared_loader_life_support_data() {
520  if (!PYBIND11_TLS_KEY_CREATE(loader_life_support_tls_key)) {
521  pybind11_fail("local_internals: could not successfully initialize the "
522  "loader_life_support TLS key!");
523  }
524  }
525  // We can't help but leak the TLS key, because Python never unloads extension modules.
526  };
527 
528  local_internals() {
529  auto &internals = get_internals();
530  // Get or create the `loader_life_support_stack_key`.
531  auto &ptr = internals.shared_data["_life_support"];
532  if (!ptr) {
533  ptr = new shared_loader_life_support_data;
534  }
535  loader_life_support_tls_key
536  = static_cast<shared_loader_life_support_data *>(ptr)->loader_life_support_tls_key;
537  }
538 #endif // defined(WITH_THREAD) && PYBIND11_INTERNALS_VERSION == 4
539 };
